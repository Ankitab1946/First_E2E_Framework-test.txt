partition "Data Dictionary Flow" #EAF3FF {

  #D9EAF7:Open Data Dictionary tab;

  #D9EAF7:User selects filters;
  note right
Portfolio
PRJ_ID
Attribute Name
Attribute Description
Section
Overlapped Attribute
end note

  if (USE_API_FOR_FILTERS = true?) then (Yes)
    #FFF2CC:Call FastAPI Filter API;
    note right
POST
/api/v1/dictionary/filter

API Port: 8502
end note
  else (No / Fallback)
    #FFF2CC:Call local Dictionary Service;
  endif

  #E2F0D9:Repository builds SQLAlchemy query;

  #EDEDED:Fetch filtered records from SQL Server;
  note right
Database: PRJ_DB
Table: master_dictionary
Only active records
end note

  #D9EAF7:Display records in Streamlit grid;

  if (User action?) then (Add New Attribute)
    #FCE4D6:Open Create Attribute popup;
    #FCE4D6:Validate mandatory fields;
    note right
PRJ_ID
Attribute Name
Section
end note
    #E2F0D9:Call Finalization Service;

  elseif (View/Edit)
    #D9EAF7:Select row from grid;
    #FCE4D6:Open Edit Attribute popup;
    #FCE4D6:Open in read-only mode;
    #FCE4D6:User clicks Edit;
    #FCE4D6:Fields become editable except PRJ_ID;
    #E2F0D9:Call Finalization Service;

  elseif (Soft Delete)
    #D9EAF7:Select row from grid;
    #F4CCCC:Set is_active = 0;
    note right
master_dictionary
prj_attribute
prj_attr_business_logic
prj_attr_business_logic_scope
end note

  elseif (Reactivate)
    #D9EAF7:View soft-deleted records;
    #D9EAF7:Select inactive row from grid;
    #E2F0D9:Click Activate Record;
    #E2F0D9:Set is_active = 1;
    note right
master_dictionary
prj_attribute
prj_attr_business_logic
prj_attr_business_logic_scope
end note

  elseif (Save to S3)
    #E2F0F9:Fetch latest active data from 4 tables;
    #E2F0F9:Create timestamped CSV files;
    #E2F0F9:Upload files to internal S3 bucket;
    note right
boto3 upload_file
endpoint_url = HOST
use_ssl = False
verify = False
end note
  endif

  #E2F0D9:Write / update target tables;
  note right
master_dictionary
prj_attribute
prj_attr_business_logic
prj_attr_business_logic_scope
end note

  #E2F0D9:Write audit and history;
  note right
audit_log
history_log
User
Timestamp
Action Type
Old Value
New Value
end note

  #EDEDED:Clear Streamlit cache;
  #D9EAF7:Refresh main grid;
}
